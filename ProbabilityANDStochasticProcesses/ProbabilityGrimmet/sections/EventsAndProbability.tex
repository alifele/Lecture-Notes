\section{Events as Sets}

\begin{problem}
	Let $ \set{A_i: i\in I} $ be a collection of sets. Prove ``De Morgan's Laws'''
	\[ \left( \bigcup_i A_i \right)^c = \bigcap_i A_i^c, \qquad \left( \bigcap_i A_i \right)^c = \bigcup_i A_i^c. \]
\end{problem}

\begin{solution}
	Let $ x \in  \left( \cup_i A_i \right)^c $. Then $ x\notin \cup_i A_i $. So $ \forall i \in I $ we have $ x\notin A_i $, or equivalently $ x\in A_i^c $. Using the definition of intersection $ x\in \cap_i A_i^c $. For the second identity,
	\begin{align*}
		& x \in  \left( \bigcap_i A_i \right)^c \qquad \biImp \\
		& x \notin \bigcap_i A_i \qquad \biImp \\
		& \exists i\in I \st x \notin A_i \qquad \biImp \\
		& \exists i\in I \st x \in A_i^c  \qquad \biImp \\
		& x \in \bigcup_i A_i^c.
	\end{align*}
\end{solution}


\begin{problem}
	Let $ A $ and $ B $ belong to some $ \sigma $-filed $ \mathcal{F} $. Show that $ \mathcal{F} $ contains the sets $ A\cap B $, $ A\backslash B $ and $ A\Delta B $.
\end{problem}
\begin{solution}
	Since $ A,B \in\mathcal{F} $, $ A^c, B^c \in \mathcal{F} $ as well. So
	\[ A\cap B = (A^c \cup B^c)^c \in \mathcal{F}, \]
	since $ A^c, B^c \in \mathcal{F} $ and $ \mathcal{F} $ is closed under union and complement. Furthermore
	\[ A\backslash B = A\cap B^c \in \mathcal{F}. \]
	And
	\[ A\Delta B = (A\cup B) \backslash (A\cap B)  \in \mathcal{F}. \]
\end{solution}

\begin{problem}
	Let $ \mathcal{F} $ be a $ \sigma $-field of subset of $ \omega $ and suppose that $ B\in\mathcal{F} $. Show that $ \mathcal{G} = \set{A\cap B: A\in\mathcal{F}}$ is a $ \sigma $-field of subsets of $ B $.
\end{problem}
\begin{solution}
	Any element of $ \mathcal{G} $ is of the form $ A\cap B $ for some $ A\in \mathcal{F} $, thus $ A\cap B \subset B $. It remains to show that $ G $ is a $ \sigma $-field.
	\begin{enumerate}[(a)]
		\item Contains $ B $ and $ \emptyset $: Because $ \Omega \in\mathcal{F} $, so $ \mathcal{G} \ni \Omega \cap B = B $. Furthermore, $ \emptyset\in\mathcal{F} $. So $ \mathcal{G} \ni \emptyset \cap B = \emptyset $.
		\item Is closed under countable union: Let $ (A_i\cap B) \in\mathcal{G} $. Then
		\[ \bigcup_i (A_i\cap B) = (\bigcup_i A_i) \cap B \in\mathcal{G}. \]
		\item Is closed under complement with respect to $ B $: Let $ A\cap B \in \mathcal{G} $. The complement of $ A\cap B $ with respect to $ B $ is
		\[ B\backslash(A\cap B) = B\cap(A\cap B) = B\cap(A^c\cup B) = A^c\cap B \in \mathcal{G}. \]
	\end{enumerate}
\end{solution}
